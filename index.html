#include<stdio.h>

#define M 5
void funkcja(int tab[M]){
int i, j, temp;
    for (i = 0; i<M; i++)
    {
            for (j=0; j<M-1-i; j++)
            {
                    if (tab[j] < tab[j+1])
                    {
                            temp = tab[j+1];
                            tab[j+1] = tab[j];
                            tab[j] = temp;
                    }
            }
    }
}
int main()
{
int tab[M];
int i;
for(i=0; i< M; i=i+1){
      tab[i]=0;
}    
printf("podaj 5 liczb naturalnych\n");
for(i=0; i< M; i=i+1){
      scanf("%d",&tab[i]);
}   
funkcja(tab);

      printf("%d\n",tab[i]);

}
